 195    """Validates moves and walls in the game."""

 197    def __init__(self, grid, walls, players):
 198        """
 199        Initializes the validator.
 200
 201        Args:\n
 202            grid (list): The game grid.
 203            walls (list): The list of walls in the game.
 204            players (list): The list of players in the game.
 205        """
 206        self.visited = set()
 207        self.grid = grid
 208        self.walls = walls

 214    def possible_moves(self, pos):
 215        """
 216        Gets possible moves for a position.
 217
 218        Args:\n
 219            pos (tuple): The current position.
 220
 221        Returns:\n
 222            list: The list of possible move positions.
 223        """
 224        ret = []
 225        for dx, dy in DEFAULT:
 226            x, y = pos
 227            if 0 <= x + dx < 9 and 0 <= y + dy < 9:
 228                if self.grid[x + dx][y + dy] is not None:
 229                    if (
 230                        0 <= x + dx * 2 < 9
 231                        and 0 <= y + dy * 2 < 9
 232                        and self.grid[x + dx * 2][y + dy * 2] is None
 233                        and not self.any_overlap((x, y), (x + 2 * dx, y + 2 * dy), True)
 234                    ):
 235                        ret.append((x + 2 * dx, y + 2 * dy))
 236                    else:
 237                        if (
 238                            dy != 0
 239                            and x + 1 < 9
 240                            and self.grid[x + 1][y + dy] is None
 241                            and not self.any_overlap((x, y), (x + 1, y + dy), True)
 242                        ):
 243                            ret.append((x + 1, y + dy))
 244                        if (
 245                            dy != 0
 246                            and x - 1 >= 0
 247                            and self.grid[x - 1][y + dy] is None
 248                            and not self.any_overlap((x, y), (x - 1, y + dy), True)
 249                        ):
 250                            ret.append((x - 1, y + dy))
 251                        if (
 252                            dx != 0
 253                            and y + 1 < 9
 254                            and self.grid[x + dx][y + 1] is None
 255                            and not self.any_overlap((x, y), (x + dx, y + 1), True)
 256                        ):
 257                            ret.append((x + dx, y + 1))
 258                        if (
 259                            dx != 0
 260                            and y - 1 >= 0
 261                            and self.grid[x + dx][y - 1] is None
 262                            and not self.any_overlap((x, y), (x + dx, y - 1), True)
 263                        ):
 264                            ret.append((x + dx, y - 1))
 265                elif not self.any_overlap((x, y), (x + dx, y + dy), True):
 266                    ret.append((x + dx, y + dy))
 267        return ret

 269    def any_overlap(self, start, end, fake=False):
 270        """
 271        Checks if any wall overlaps between two positions.
 272
 273        Args:\n
 274            start (tuple): The start position.

 276            fake (bool, optional): If the wall is fake. Defaults to False.

 282        for wall in self.walls:

 284                return True

 307    def possible_orients(self, pos):
 308        """
 309        Returns list of all possible orientations of wall in current position.
 310
 311        Args:\n
 312            pos (tuple): Start of the wall to check. 
 313        Returns:\n
 314            list: tuple list of all directions wall can face in give position.
 315        """
 316        ret = []
 317        for dx, dy in DEFAULT:
 318            end = (pos[0] + 2 * dx, pos[1] + 2 * dy)
 319            if -1 <= end[0] < 9 and -1 <= end[1] < 9:
 320                if not self.any_overlap(pos, end):

 333    def way_exists(self, node, dest):
 334        """
 335        Depth-first search to find connection between nodes.
 336
 337        Args:\n
 338            node (tuple): The current node.
 339            dest (tuple): The destination node.
 340        Returns:\n
 341            bool: True if the end node is reachable, False otherwise.
 342        """
 343
 344        if node[0] == dest[0] or node[1] == dest[1]:
 345            return True
 346        else:
 347            # print(node)
 348            self.visited.add(node)
 349            for dx, dy in DEFAULT:
 350                end = (node[0] + dx, node[1] + dy)
 351                if (
 352                    0 <= end[0] < 9
 353                    and 0 <= end[1] < 9
 354                    and end not in self.visited
 355                    and not self.any_overlap(node, end, True)
 356                ):
 357                    if self.way_exists(end, dest):
 358                        return True
 359            return False

 361    def bfs(self, pos, wins):
 362        """
 363        Breadth-first search to find shortest way between nodes.
 364
 365        Args:\n
 366            pos (tuple): The current position.
 367            wins (tuple): The destination nodes.
 368        Returns:\n
 369            int: Length of shortest path to win.
 370        """
 371        queue = deque()
 372        self.reset()
 373        queue.append((pos, 0))
 374        self.visited.add(pos)
 375        while queue:
 376            curr, depth = queue.popleft()
 377
 378            if curr[0] == wins[0] or curr[1] == wins[1]:
 379                return depth
 380            for move in self.possible_moves(curr):
 381                if move not in self.visited:
 382                    self.visited.add(move)
 383                    queue.append((move, depth + 1))
 384        return 1000000

 386    def best_move(self, start, wins):
 387        """
 388        Determines the best move for the player.
 389
 390        Args:\n
 391            start (tuple): The start position.
 392            wins (tuple): The wining positions.
 393
 394        Returns:\n
 395            tuple: The best move for the player.
 396        """
 397        moves = self.possible_moves(start)
 398        bmove, bscore = moves[0], self.bfs(moves[0], wins)
 399        for move in moves:
 400            score = self.bfs(move, wins)
 401            if score < bscore:
 402                bmove = move
 403                bscore = score
 404        return bmove, bscore

 406    def all_possible_walls(self):
 407        """
 408        Returns list of all possible walls.
 409
 410        Args:\n
 411
 412        Returns:\n
 413            list: tuple list of all possible starts of wall.
 414        """
 415        ret = []
 416        for y in range(8):
 417            for x in range(8):
 418                for dist in self.possible_orients((x, y)):
 419                    ret.append(((x, y), (dist)))

 422    def best_wall(self, my_pos, win):
 423        """
 424        Determines the best wall placement for the player.
 425
 426        Args:\n
 427            my_pos (tuple): Position of my player.
 428            win (tuple): The winning positions.
 429
 430        Returns:\n
 431            tuple: The best wall placement for player.
 432        """
 433        all_w = self.all_possible_walls()
 434        samples = all_w
 435        bwall, bscore = None, -1000
 436
 437        for pos, dir in samples:
 438            score = 0
 439            self.walls.append(Wall(pos, (pos[0] + 2 * dir[0], pos[1] + 2 * dir[1])))
 440            for player in self.players:
 441                if player.position != my_pos:
 442                    # print(pos,player.position)
 443                    score += self.bfs(player.position, player.wins)
 444            score -= self.bfs(my_pos, win)
 445            del self.walls[-1]
 446            if score > bscore:
 447                bwall = (pos, dir)
 448                bscore = score
 449        return bwall
